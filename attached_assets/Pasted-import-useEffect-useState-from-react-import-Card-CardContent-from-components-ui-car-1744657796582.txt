import { useEffect, useState } from "react";
import { Card, CardContent } from "@/components/ui/card";

export default function IntelligenceStatusBanner({ sessionId }) {
  const [status, setStatus] = useState(null);

  useEffect(() => {
    const fetchStatus = async () => {
      const res = await fetch(`/api/session/intelligence-indicators/${sessionId}`);
      const data = await res.json();
      setStatus(data);
    };
    fetchStatus();
  }, [sessionId]);

  if (!status) return null;

  const items = [
    ["📄 IND Summary", "ind_summary", "ind_module_with_context.docx"],
    ["📊 Summary Packet", "summary_packet", "summary_packet.pdf"],
    ["📉 Dropout Forecast", "risk_model", "dropout_forecast.json"],
    ["📈 Success Prediction", "success_model", "success_prediction.json"],
    ["🛠 Suggestions", "suggestions", "suggested_corrections.json"],
    ["📎 CSR Alignment", "alignment_score", "alignment_score_report.json"],
    ["📘 SAP Document", "sap_summary", "sap_summary_branded.docx"],
    ["🧠 Assistant Trace", "wisdom_trace", "wisdom_trace.json"],
    ["📽 Strategy Deck", "strategy_slide", "trial_strategy_deck.pptx"]
  ];

  return (
    <Card className="border-l-4 border-indigo-500 bg-indigo-50 mb-6">
      <CardContent className="p-4 space-y-1">
        <h2 className="text-sm font-semibold text-indigo-900 mb-2">📋 Intelligence Module Status</h2>
        <ul className="text-sm text-indigo-900 space-y-1">
          {items.map(([label, key, file], i) => (
            <li key={i}>
              {status[key] ? (
                <>
                  ✅ {label} –{" "}
                  <a
                    href={`/static/lumen_reports_backend/sessions/${sessionId}/${file}`}
                    className="underline hover:text-indigo-600"
                    target="_blank"
                    rel="noopener noreferrer"
                  >
                    View
                  </a>
                </>
              ) : (
                <>❌ {label} – Not yet generated</>
              )}
            </li>
          ))}
        </ul>
      </CardContent>
    </Card>
  );
}
