      - name: Checkout code

      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: '18.x'

      - name: Install dependencies and build frontend
        run: |
          cd client
          npm install
          npm run build

      - name: Install backend dependencies
        run: |
          cd server
          npm install

      - name: Deploy
        run: echo "Deploy script goes here (e.g., upload to cloud or server)"
